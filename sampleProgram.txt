class AddNum
{
	public int addNumber( int a, int b)
	{
		int total, firstInt, secondInt;
		{

			total = 0;
			firstInt = a;
			secondInt = b;

			do
			{
				{
					total = firstInt + secondInt;
					System.out.println(total);
				}
			}
			while ( total < 100);

			return total;

		}
	}
